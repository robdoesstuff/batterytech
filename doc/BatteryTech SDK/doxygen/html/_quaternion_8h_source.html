542  Quaternion<T> slerp(T r, const Quaternion<T>& q2) const {
543  Quaternion<T> ret;
544  T cosTheta = w * q2.w + v.x * q2.v.x + v.y * q2.v.y + v.z * q2.v.z;
545  T theta = (T) acos(cosTheta);
546  if (fabs(theta) < epsilon) {
547  ret = *this;
548  } else {
549  T sinTheta = (T) sqrt(1.0 - cosTheta * cosTheta);
550  if (fabs(sinTheta) < epsilon) {
551  ret.w = 0.5 * w + 0.5 * q2.w;
552  ret.v = v.lerp(0.5, q2.v);
553  } else {
554  T rA = (T) sin((1.0 - r) * theta) / sinTheta;
555  T rB = (T) sin(r * theta) / sinTheta;
556 
557  ret.w = w * rA + q2.w * rB;
558  ret.v.x = v.x * rA + q2.v.x * rB;
559  ret.v.y = v.y * rA + q2.v.y * rB;
560  ret.v.z = v.z * rA + q2.v.z * rB;
561  }
562  }
563  return ret;
564  }
